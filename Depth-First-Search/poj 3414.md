<h1><strong>Pots</strong></h1>

### **Description**
You are given two pots, having the volume of A and B liters respectively. The following operations can be performed:
1. FILL(i)        fill the pot i (1 ≤ i ≤ 2) from the tap;
2. DROP(i)      empty the pot i to the drain;
3. POUR(i,j)    pour from pot i to pot j; after this operation either the pot j is full (and there may be some water left in the pot i), or the pot i is empty (and all its contents have been moved to the pot j).

Write a program to find the shortest possible sequence of these operations that will yield exactly C liters of water in one of the pots.

### **Input**

On the first and only line are the numbers A, B, and C. These are all integers in the range from 1 to 100 and C≤max(A,B).

### **Output**
The first line of the output must contain the length of the sequence of operations K. The following K lines must each describe one operation. If there are several sequences of minimal length, output any one of them. If the desired result can’t be achieved, the first and only line of the file must contain the word <strong>'impossible'</strong>

### **Sample Input**
3 5 4<br/>

### **Sample Output**
6<br/>
FILL(2)<br/>
POUR(2,1)<br/>
DROP(1)<br/>
POUR(2,1)<br/>
FILL(2)<br/>
POUR(2,1)<br/>

#include <iostream>
#include <queue>
#include <stack>
#include <string>
using namespace std;

bool visit[101][101];

class node
{
public:
	int i;
	int j;
	node *lastnode;
	string operation;
	int step;
};
node* tailnode;

int* poured(int i, int j, int rongliangi, int rongliangj)
{
	int *result = new int[2];
	if (i + j >= rongliangj)
	{
		result[0] = i + j - rongliangj;
		result[1] = rongliangj;
	}
	else
	{
		result[0] = 0;
		result[1] = i + j;
	}
	return result;
}

bool bfs(int i, int j, int k)
{
	//初始化
	node *begin = new node();
	begin->i = 0;
	begin->j = 0;
	begin->step = 0;
	begin->lastnode = NULL;
	visit[begin->i][begin->j] = true;
	queue<node*> q;
	q.push(begin);

	while (!q.empty())
	{
		node* n = q.front();
		q.pop();
		if (n->i == k || n->j == k)
		{
			cout << n->step;
			tailnode = n;
			stack<node> s;
			node* temp = tailnode;
			while (temp != NULL)
			{
				s.push(*temp);
				temp = temp->lastnode;
			}
				
			while (!s.empty())
			{
				cout << s.top().operation << endl;
				s.pop();
			}			
			return true;
		}
		//FILL(i)
		if (!visit[i][n->j])
		{
			node *pNew = new node();
			pNew->lastnode = n;
			pNew->operation = "FILL(1)";
			pNew->i = i;
			pNew->j = n->j;
			visit[i][n->j] = true;
			pNew->step = n->step + 1;
			q.push(pNew);
		}

		//FILL(j)

		if (!visit[n->i][j])
		{
			node *pNew = new node();
			pNew->i = n->i;
			pNew->j = j;
			pNew->lastnode = n;
			pNew->operation = "FILL(2)";
			visit[n->i][j] = true;
			pNew->step = n->step + 1;
			q.push(pNew);
		}
		//POUR(1,2)
		int* s = new int[2];
		s = poured(n->i, n->j, i, j);
		if (!visit[s[0]][s[1]])
		{
			node *pNew = new node();
			pNew->i = s[0];
			pNew->j = s[1];
			pNew->lastnode = n;
			pNew->operation = "POUR(1,2)";
			visit[s[0]][s[1]] = true;
			pNew->step = n->step + 1;
			q.push(pNew);
		}
		//POUR(2,1)
		int* s2 = new int[2];
		s2 = poured(n->j, n->i, j, i);
		if (!visit[s2[1]][s2[0]])
		{
			node *pNew = new node();
			pNew->i = s2[1];
			pNew->j = s2[0];
			pNew->lastnode = n;
			pNew->operation = "POUR(2,1)";
			visit[s2[0]][s2[1]] = true;
			pNew->step = n->step + 1;
			q.push(pNew);
		}

		//DROP(i)
		if (!visit[0][n->j])
		{
			node *pNew = new node();
			pNew->i = 0;
			pNew->j = n->j;
			pNew->lastnode = n;
			pNew->operation = "DROP(1)";
			visit[0][n->j] = true;
			pNew->step = n->step + 1;
			q.push(pNew);
		}
		//DROP(j)
		if (!visit[n->i][0])
		{
			node *pNew = new node();
			pNew->i = n->i;
			pNew->j = 0;
			pNew->lastnode = n;
			pNew->operation = "DROP(2)";
			visit[n->i][0] = true;
			pNew->step = n->step + 1;
			q.push(pNew);
		}
	}
	return false;
}

int main()
{
	int pot1, pot2, pot3;
	cin >> pot1 >> pot2 >> pot3;
	if (bfs(pot1, pot2, pot3) == 0)
		cout << "impossible";

	return 0;
}
